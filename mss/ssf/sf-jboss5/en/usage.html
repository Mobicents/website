sf-jboss5

Usage

SSF is available on JBoss5, if you define sip.xml as follows.

<?xml version="1.0" encoding="UTF-8"?>
<sip-app xmlns="http://www.jcp.org/xml/ns/sipservlet"
           xmlns:javaee="http://java.sun.com/xml/ns/javaee">
  ...
  <servlet>
    <javaee:servlet-name>sip-dispatcher</javaee:servlet-name>
    <javaee:servlet-class>com.oki.sip.sf.servlet.SipDispatcherServlet</javaee:servlet-class>
    <javaee:load-on-startup>10</javaee:load-on-startup>
    <init-param>
      <param-name>contextClass</param-name>
      <param-value>org.mobicents.ssf.context.spring.support.VfsSupportXmlSipApplicatoinContext</param-value>
    </init-param>
  </servlet>
  
  ...
</sip-app>

Similarly, SpringFramework (2.5.x) is available on JBoss 5, if you define web.xml as follows.

   ...
   <context-param>
    <description>
    This parameter is a setting for supporting JBoss VFS.
    If you are not using the JBoss 5, please comment out this setting.
    </description>
    <param-name>contextClass</param-name>
    <param-value>
      org.mobicents.ssf.context.spring.support.VfsSupportXmlWebApplicationContext
    </param-value>
  </context-param>
  
  ...
  
  <servlet>
    <servlet-name>dispatcher</servlet-name>
    <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
    <init-param>
      <param-name>contextClass</param-name>
      <param-value>
        org.mobicents.ssf.context.spring.support.VfsSupportXmlWebApplicationContext
      </param-value>
    </init-param>    
    <load-on-startup>2</load-on-startup>
  </servlet>

  ...
